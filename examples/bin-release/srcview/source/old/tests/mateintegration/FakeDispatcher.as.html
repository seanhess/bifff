            this.dispatchEvent(new FlexEvent(FlexEvent.CREATION_COMPLETE, true));
        }
        
        public function go():void
        {
            var event:Event = new Event("test", true);
            dispatchEvent(event);
        }

        public var id:String;
        
    }
